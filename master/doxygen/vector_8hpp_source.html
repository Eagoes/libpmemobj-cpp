494 template <typename T>
495 void
496 vector<T>::assign(size_type count, const_reference value)
497 {
498  pool_base pb = get_pool();
499 
500  transaction::run(pb, [&] {
501  if (count <= capacity()) {
502  /*
503  * Reallocation is not needed. First, replace old
504  * elements with new ones in range [0, size()).
505  * Depending on count, either call remaining old
506  * elements destructors, or append more new elements.
507  */
508  size_type size_old = _size;
509  snapshot_data(0, size_old);
510 
511  std::fill_n(
512  &_data[0],
513  (std::min)(count,
514  static_cast<size_type>(size_old)),
515  value);
516 
517  if (count > size_old) {
518 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
519  /*
520  * Range of memory:
521  * [&_data[size_old], &_data[count])
522  * is undefined, there is no need to snapshot
523  * and eventually rollback old data.
524  */
525  VALGRIND_PMC_ADD_TO_TX(
526  &_data[size_old],
527  sizeof(T) * (count - size_old));
528 #endif
529 
530  construct(size_old, count - size_old, value);
531  /*
532  * XXX: explicit persist is required here
533  * because given range wasn't snapshotted and
534  * won't be persisted automatically on tx
535  * commit. This can be changed once we will have
536  * implemented "uninitialized" flag for
537  * pmemobj_tx_xadd in libpmemobj.
538  */
539  pb.persist(&_data[size_old],
540  sizeof(T) * (count - size_old));
541  } else {
542  shrink(count);
543  }
544  } else {
545  dealloc();
546  alloc(count);
547  construct(0, count, value);
548  }
549  });
550 }

570 template <typename T>
571 template <typename InputIt,
572  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
573  InputIt>::type *>
574 void
575 vector<T>::assign(InputIt first, InputIt last)
576 {
577  pool_base pb = get_pool();
578 
579  size_type size_new = static_cast<size_type>(std::distance(first, last));
580 
581  transaction::run(pb, [&] {
582  if (size_new <= capacity()) {
583  /*
584  * Reallocation is not needed. First, replace old
585  * elements with new ones in range [0, size()).
586  * Depending on size_new, either call remaining old
587  * elements destructors, or append more new elements.
588  */
589  size_type size_old = _size;
590  snapshot_data(0, size_old);
591 
592  InputIt mid = last;
593  bool growing = size_new > size_old;
594 
595  if (growing) {
596 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
597  /*
598  * Range of memory:
599  * [&_data[size_old], &_data[size_new])
600  * is undefined, there is no need to snapshot
601  * and eventually rollback old data.
602  */
603  VALGRIND_PMC_ADD_TO_TX(
604  &_data[size_old],
605  sizeof(T) * (size_new - size_old));
606 #endif
607 
608  mid = first;
609  std::advance(mid, size_old);
610  }
611 
612  iterator shrink_to = std::copy(first, mid, &_data[0]);
613 
614  if (growing) {
615  construct_range_copy(size_old, mid, last);
616  /*
617  * XXX: explicit persist is required here
618  * because given range wasn't snapshotted and
619  * won't be persisted automatically on tx
620  * commit. This can be changed once we will have
621  * implemented "uninitialized" flag for
622  * pmemobj_tx_xadd in libpmemobj.
623  */
624  pb.persist(&_data[size_old],
625  sizeof(T) * (size_new - size_old));
626  } else {
627  shrink(static_cast<size_type>(std::distance(
628  iterator(&_data[0]), shrink_to)));
629  }
630  } else {
631  dealloc();
632  alloc(size_new);
633  construct_range_copy(0, first, last);
634  }
635  });
636 }

1584 void
1585 vector<T>::insert_gap(size_type idx, size_type count)
1586 {
1587  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
1588 
1589  if (_capacity >= _size + count) {
1590  pointer dest = &_data[size() + count];
1591  pointer begin = &_data[idx];
1592  pointer end = &_data[size()];
1593 
1594  /*
1595  * XXX: There is no necessity to snapshot uninitialized data, so
1596  * we can optimize it by calling:
1597  * transaction::snapshot<T>(begin, size() - idx).
1598  * However, we need libpmemobj support for that, because right
1599  * now pmemcheck will report an error (uninitialized part of
1600  * data not added to tx).
1601  *
1602  * XXX: future optimization: we don't have to snapshot data
1603  * which we will not overwrite
1604  */
1605 #if LIBPMEMOBJ_CPP_VG_MEMCHECK_ENABLED
1606  VALGRIND_MAKE_MEM_DEFINED(end, sizeof(T) * count);
1607 #endif
1608  transaction::snapshot<T>(begin, size() + count);
1609 
1610  std::move_backward(begin, end, dest);
1611  } else {
1612  /*
1613  * XXX: future optimization: we don't have to snapshot data
1614  * which we will not overwrite
1615  */
1616  snapshot_data(0, _size);
1617 
1618  auto old_data = _data;
1619  auto old_size = _size;
1620  pointer old_begin = &_data[0];
1621  pointer old_mid = &_data[idx];
1622  pointer old_end = &_data[size()];
1623 
1624  _data = nullptr;
1625  _size = _capacity = 0;
1626 
1627  alloc(get_recommended_capacity(old_size + count));
1628 
1629  construct_range(0, old_begin, old_mid);
1630  construct_range(idx + count, old_mid, old_end);
1631 
1632  /* destroy and free old data */

1634  detail::destroy<value_type>(old_data[i]);

1636  throw transaction_free_error(
