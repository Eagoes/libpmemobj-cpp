583 template <typename T>
584 void
585 vector<T>::assign(size_type count, const_reference value)
586 {
587  pool_base pb = get_pool();
588 
589  transaction::run(pb, [&] {
590  if (count <= capacity()) {
591  /*
592  * Reallocation is not needed. First, replace old
593  * elements with new ones in range [0, size()).
594  * Depending on count, either call remaining old
595  * elements destructors, or append more new elements.
596  */
597  size_type size_old = _size;
598  snapshot_data(0, size_old);
599 
600  std::fill_n(
601  &_data[0],
602  (std::min)(count,
603  static_cast<size_type>(size_old)),
604  value);
605 
606  if (count > size_old) {
607 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
608  /*
609  * Range of memory:
610  * [&_data[size_old], &_data[count])
611  * is undefined, there is no need to snapshot
612  * and eventually rollback old data.
613  */
614  VALGRIND_PMC_ADD_TO_TX(
615  &_data[static_cast<difference_type>(
616  size_old)],
617  sizeof(T) * (count - size_old));
618 #endif
619 
620  construct(size_old, count - size_old, value);
621  /*
622  * XXX: explicit persist is required here
623  * because given range wasn't snapshotted and
624  * won't be persisted automatically on tx
625  * commit. This can be changed once we will have
626  * implemented "uninitialized" flag for
627  * pmemobj_tx_xadd in libpmemobj.
628  */
629  pb.persist(&_data[static_cast<difference_type>(
630  size_old)],
631  sizeof(T) * (count - size_old));
632  } else {
633  shrink(count);
634  }
635  } else {
636  dealloc();
637  alloc(count);
638  construct(0, count, value);
639  }
640  });
641 }

661 template <typename T>
662 template <typename InputIt,
663  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
664  InputIt>::type *>
665 void
666 vector<T>::assign(InputIt first, InputIt last)
667 {
668  pool_base pb = get_pool();
669 
670  size_type size_new = static_cast<size_type>(std::distance(first, last));
671 
672  transaction::run(pb, [&] {
673  if (size_new <= capacity()) {
674  /*
675  * Reallocation is not needed. First, replace old
676  * elements with new ones in range [0, size()).
677  * Depending on size_new, either call remaining old
678  * elements destructors, or append more new elements.
679  */
680  size_type size_old = _size;
681  snapshot_data(0, size_old);
682 
683  InputIt mid = last;
684  bool growing = size_new > size_old;
685 
686  if (growing) {
687 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
688  /*
689  * Range of memory:
690  * [&_data[size_old], &_data[size_new])
691  * is undefined, there is no need to snapshot
692  * and eventually rollback old data.
693  */
694  VALGRIND_PMC_ADD_TO_TX(
695  &_data[static_cast<difference_type>(
696  size_old)],
697  sizeof(T) * (size_new - size_old));
698 #endif
699 
700  mid = first;
701  std::advance(mid, size_old);
702  }
703 
704  iterator shrink_to = std::copy(first, mid, &_data[0]);
705 
706  if (growing) {
707  construct_range_copy(size_old, mid, last);
708  /*
709  * XXX: explicit persist is required here
710  * because given range wasn't snapshotted and
711  * won't be persisted automatically on tx
712  * commit. This can be changed once we will have
713  * implemented "uninitialized" flag for
714  * pmemobj_tx_xadd in libpmemobj.
715  */
716  pb.persist(&_data[static_cast<difference_type>(
717  size_old)],
718  sizeof(T) * (size_new - size_old));
719  } else {
720  shrink(static_cast<size_type>(std::distance(
721  iterator(&_data[0]), shrink_to)));
722  }
723  } else {
724  dealloc();
725  alloc(size_new);
726  construct_range_copy(0, first, last);
727  }
728  });
729 }

2168 template <typename T>
2169 void
2170 vector<T>::insert_gap(size_type idx, size_type count)
2171 {
2172  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
2173 
2174  if (_capacity >= _size + count) {
2175  pointer dest =
2176  &_data[static_cast<difference_type>(size() + count)];
2177  pointer begin = &_data[static_cast<difference_type>(idx)];
2178  pointer end = &_data[static_cast<difference_type>(size())];
2179 
2180  /*
2181  * XXX: There is no necessity to snapshot uninitialized data, so
2182  * we can optimize it by calling:
2183  * transaction::snapshot<T>(begin, size() - idx).
2184  * However, we need libpmemobj support for that, because right
2185  * now pmemcheck will report an error (uninitialized part of
2186  * data not added to tx).
2187  *
2188  * XXX: future optimization: we don't have to snapshot data
2189  * which we will not overwrite
2190  */
2191 #if LIBPMEMOBJ_CPP_VG_MEMCHECK_ENABLED
2192  VALGRIND_MAKE_MEM_DEFINED(end, sizeof(T) * count);
2193 #endif
2194  snapshot_data(idx, _size + count);
2195 
2196  std::move_backward(begin, end, dest);
2197  } else {
2198  /*
2199  * XXX: future optimization: we don't have to snapshot data
2200  * which we will not overwrite
2201  */
2202  snapshot_data(0, _size);
2203 
2204  auto old_data = _data;
2205  auto old_size = _size;
2206  pointer old_begin = &_data[0];
2207  pointer old_mid = &_data[static_cast<difference_type>(idx)];
2208  pointer old_end = &_data[static_cast<difference_type>(size())];
2209 
2210  _data = nullptr;
2211  _size = _capacity = 0;
2212 
2213  alloc(get_recommended_capacity(old_size + count));
2214 
2215  construct_range(0, old_begin, old_mid);
2216  construct_range(idx + count, old_mid, old_end);
2217 
2218  /* destroy and free old data */
2219  for (size_type i = 0; i < old_size; ++i)
2220  detail::destroy<value_type>(
2221  old_data[static_cast<difference_type>(i)]);
